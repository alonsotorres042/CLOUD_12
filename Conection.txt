function addUser($username, $password, $creator) {
    global $conn;

    $stmt = $conn->prepare("CALL AddUser(?, ?, ?)");
    $stmt->bind_param("sss", $username, $password, $creator);

    if ($stmt->execute()) {
        echo "User added successfully";
    } else {
        echo "Error: " . $stmt->error;
    }

    $stmt->close();
}
